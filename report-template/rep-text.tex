\section{Второй раздел}

\subsection{Подраздел}


Первая чудоформула
\begin{equation}\label{eq:H}
H_0 = \frac{p^2}{2 m} + a_0\, x^4 - a_2\, x^2,
\end{equation}
 а это чудогамильтониан записанный в базисе чудогармонических осцилляторов
\begin{align}
H^0_{m\, n} & = \delta_{m + 4 \; n} \;\frac{a_0 g^2}{4} \sqrt{(m + 1)(m + 2)(m + 3)(m +
4)} \notag\\
 & + \delta_{m + 2 \; n} \;\frac{g}{2} \left(g\, a_0  (2 m + 3) - a'_2 \right) \sqrt{(m
+ 1)(m  + 2)}.
\end{align}
